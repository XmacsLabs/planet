配置用户名和邮箱

    

      警告.
      请使用你自己的用户名和邮箱
    

    

      

        

          Shell] 
          
git config --global user.name "沈浪熊猫儿"

        

      

    

    

      

        

          Shell] 
          
git config --global user.email "shenda@ustc.edu"

        

      

    

    

      

        

          
Shell] 

          
cat ~/.gitconfig

        

      

    

    

      

        

          

            [alias]
          

        

      

    

    

      

        

          

            co = checkout
          

        

      

    

    

      

        

          

            ci = commit
          

        

      

    

    

      

        

          

            br = branch
          

        

      

    

    

      

        

          

            [core]
          

        

      

    

    

      

        

          

            quotepath = false
          

        

      

    

    

      

        

          

            [user]
          

        

      

    

    

      

        

          

            name = 沈浪熊猫儿
          

        

      

    

    

      

        

          

            email = shenda@ustc.edu
