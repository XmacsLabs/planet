(define (scm2stm x)
  (define (x->string x)
    (cond ((number? x) (number->string x))
          ((list? x) (scm2stm x))
          (else x)))
  (if (eq? () x)
      ()
      (cons (x->string (car x))
            (scm2stm (cdr x))))))

(define (expmod base exp m)
  (cond ((= exp 0) 1)
        ((even? exp)
         (remainder (square (expmod base (/ exp 2) m))
                    m))
        (else
         (remainder (* base (expmod base (- exp 1) m))
                    m))))
